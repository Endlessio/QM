# The numbers.
x4, y4 = np.linspace(-4, 4, 100), np.linspace(-4, 4, 100)
x, y = np.meshgrid(x4, y4)

# The plot.
fig = plt.figure()
ax1 = fig.add_subplot(121)
ax1.plot(x4, y4, ls='--')
ax1.plot(x4, -y4, ls='--')
ax1.contour(x, y, 2 + x**2 - y**2, [0])
ax1.contour(x, y, -2 + x**2 - y**2, [0])
ax1.set_title("Constant\nintervals")
ax1.axis('square')
ax1.set_xticklabels('')
ax1.set_yticklabels('')
plt.xlabel('dR')
plt.ylabel('dt')

ax2 = fig.add_subplot(122)
ax2.axvline(0, ls='--')
ax2.axhline(0, ls='--')
ax2.contour(x, y, 2 + x * y, [0])
ax2.contour(x, y, -2 + x * y, [0])
ax2.axis('square')
ax2.set_title("Constant\nspace-times-time")
ax2.axis('square')
ax2.set_xlabel('dR')
ax2.set_ylabel('dt')
ax2.set_xticklabels('')
ax2.set_yticklabels('')

plt.subplots_adjust(wspace=.6)
plt.savefig('constant_intervals_space_times_time.png')
plt.show()
